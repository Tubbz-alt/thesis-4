\begin{frame}{My View on Question Answering}
\hh{Two parts to question answering:}
\begin{itemize}
  \itemsep 1em
  \item Understanding the question
  \begin{itemize}
    \itemsep 1em
    \item<2-> Most Semantic Parsing work seems to be here
    \item<2-> Can be arbitrarily hard; I only worry about easy factoids
  \end{itemize}

  \item \textcolor<3->{darkgreen}{Understanding the corpus (IE)}
  \begin{itemize}
    \itemsep 1em
    \item<4-> What facts are in the corpus? 
      \cite{key:2014angeli-kbp,key:2014angeli-active}
    \item<5-> What facts are entailed by the corpus? \newline
      [ACL OpenIE Submission]
    \item<6-> How do we find if a fact is entailed by the corpus?
      \cite{key:2013angeli-truth,key:2014angeli-naturalli}
  \end{itemize}
\end{itemize}
\end{frame}

\hspace{2em}\darkgreen{\checkmark} \w{Where was Barack Obama born?} \\
\pause
\hspace{2em}\textcolor{orange}{?} \w{What treatments are there for diabetes?} \\
\hspace{2em}\darkred{\xmark} \w{Do cats have tails?} \\

\hspace{2em}\darkred{\xmark} \w{All animals have tails.} \\
